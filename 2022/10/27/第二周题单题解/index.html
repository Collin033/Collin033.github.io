A - 加工生产调度

题意

中文题面且不长，这个应该不用解释了吧

思路

这道题是一道贪心题，由题意每个产品都先进入A车间加工后进入B车间加工，那么容易得出A机器将会不停地进行作业，但B机器在加工过程中有可能要等待A机器。我们注意到两个关键的时间点：第一个部件在A机器上加工时，B机器必须等待；最后一个部件在B机器上加工时，A机器也必须等待B机器的完工。

由上可以猜测结论：
A机器上加工时间最短的部件最先加工，这样使得B机器能在最短的空闲时间内开始加工；把在B机器上加工时间最短的部件放在最后加工，这样使得A机器用最短时间等待B机器完工。
（是不是在想这也没有证明出来啊？这就涉及到技巧问题了，在ACM竞赛中经常有贪心的题目可以大胆猜测结论，并列举多点样例进行验证，不用严格证明，甚至有些题目能利用样例的规律推到公式结论得出答案）

严格证明可看【结论】加工生产调度

代码：

#include<bits/stdc++.h>
#define int long long
using namespace std;

const int N = 1e3 + 10;

int a[N], b[N];

int n;

struct node {
    int val;
    int loc;
}c[N];

vector<node> ans;

bool cmp(node x, node y)
{
    return x.val < y.val;
}
signed main()
{
    cin >> n;
    for (int i = 1; i <= n; i++)
        cin >> a[i];
    for (int i = 1; i <= n; i++)
    {
        cin >> b[i];
        c[i].val = min(a[i], b[i]), c[i].loc = i;
    }
    sort(c + 1, c + n + 1, cmp);
    int be = 0, en = 0;
    for (int i = 1; i <= n; i++)
    {
        if (c[i].val == a[c[i].loc])
        {
            ans.insert(ans.begin()+be, c[i]);
            be++;
        }
        else
        {
            ans.insert(ans.end() - en, c[i]);
            en++;
        }
    }
    int tim = 0, tima = 0, timb = 0;
    for (int i = 0; i < ans.size(); i++)
    {
        tim += a[ans[i].loc];
        if (timb >= a[ans[i].loc])
        {
            timb -= a[ans[i].loc];
        }
        else
            timb = 0;
        timb += b[ans[i].loc];
            
    }
    cout << tim + timb << endl;
    for (int i = 0; i < ans.size(); i++)
        cout << ans[i].loc << ' ';
    return 0;
}

B - Score

题意：

给出一个由O和 X组成的串（长度为1-80），最多有十个串,统计得分。每个O的得分为目前连续出现的O的个数，X 的得分为0。如OOXXOXXOOO得分为1+2+0+0+1+0+0+1+2+3。

思路:

这题签到题，不用多说

代码：

#include<bits/stdc++.h>
#define int long long
using namespace std;

const int N = 1e3 + 10;

int t;
string str;


signed main()
{
    cin >> t;
    while (t--)
    {
        cin >> str;
        int sum = 0, temp = 0;
        for (int i = 0; i <= str.size(); i++)
        {
            if (str[i] == 'O')
                temp++;
            else
                temp = 0;
            sum += temp;
        }
        cout << sum << endl;
    }
    return 0;
}

C - Molar mass

题意：

给出分子式算其摩尔质量（没记错的话是叫这个吧）

（另外注意遇到这些题目很长的英文题目，可以跳着看，先看看样例啥的，因为前面可能会挺多情景的，像这题看样例就很直观的明白题意了）

思路：

水题，无需多言

代码：

#include<bits/stdc++.h>
#define int long long
using namespace std;

int t;
string str;

map<char, double> mp;


signed main()
{
    mp['C'] = 12.01; mp['H'] = 1.008; mp['O'] = 16.00; mp['N'] = 14.01;
    cin >> t;
    while (t--)
    {
        double sum = 0, cnt = 0, tmp = 0;
        cin >> str;
        for (int i = 0; i < str.size(); i++)
        {
            if (mp[str[i]])
            {
                if (cnt == 0)
                    sum += tmp;
                else
                    sum += tmp * cnt;
                tmp = mp[str[i]];
                cnt = 0;
            }
            else
                cnt = (cnt * 10 + (str[i] - '0'));
        }
        if (cnt == 0)
            sum += tmp;
        else
            sum += tmp * cnt;
        printf("%.3f\n",sum);
    }
    return 0;
}

D - 种树

题意：

满足n个区间种树的要求，求最少种多少棵数

思路（贪心）：

  1. 先按结束位置从小到大排序
  2. 对每个区间依次处理
    • 从前到后扫描这个区间，统计已选点的个数
    • 若已选点的个数超过了要求的点数，则continue
    • 否则从该区间由后向前扫描，添加缺少的覆盖点

代码：

#include<bits/stdc++.h>
#define int long long
using namespace std;

const int N = 3e4 + 10;

struct node {
    int b;
    int e;
    int t;
}add[N];

int n, h,ans=0;

int book[N];//标记当前段有无种树

bool cmp(node x, node y)
{
    if(x.e!=y.e)
        return x.e < y.e;
    return x.b < y.b;
}

signed main()
{
    cin >> n >> h;
    for (int i = 1; i <= h; i++)
        cin >> add[i].b >> add[i].e >> add[i].t;
    sort(add + 1, add + h + 1, cmp);
    for (int i = 1; i <= h; i++)
    {
        int cnt = 0,flag=0;
        for (int j = add[i].b; j <= add[i].e; j++)
        {
            if (book[j])
                cnt++;
            if (cnt >= add[i].t)
            {
                flag = 1;
                break;
            }
        }
        if (flag == 1)
            continue;
        for (int j = add[i].e; j >= add[i].b; j--)
        {
            if (!book[j])
            {
                book[j] = 1;
                ans++;
                cnt++;
            }
            if (cnt >= add[i].t)
                break;
        }
    }
    cout << ans << endl;
    return 0;
}

E - 喷水装置

题意：

题意不解释

思路：

要解决此题，我们得了解每个喷水装置的实际可用覆盖范围
所以思路为：
求出每个圆左右交点，因为要求出最少用到的圆，所以可以按照每个圆的左交点从小到大排序，从左到右选择所需要的圆；

代码：

#include<bits/stdc++.h>
using namespace std;

const int N = 25000 + 5;

struct node {
    double l;
    double r;
}a[N];

int t, n, w, l, loc, r;

bool cmp(node x, node y){return x.l < y.l;}

signed main()
{
    cin >> t;
    while (t--)
    {
        cin >> n >> l >> w;
        int len=0,ans=0;
        double dis;
        for (int i = 1; i <= n; i++)
        {
            cin >> loc >> r;
            if (r <= w/2)
                continue;
            dis = sqrt(r * r*1.0 - (1.0*w * w / 4));
            a[++len].l = loc - dis; a[len].r = loc + dis;
        }
        sort(a + 1, a + len + 1, cmp);
        double tmp = 0;
        int flag = 0;
        while (tmp < l)
        {
            ans++;
            double x = tmp;
            for (int i = 1; a[i].l <= x && i <= len; i++)
                if (tmp < a[i].r)
                    tmp = a[i].r;
            if (tmp == x && x < l)
            {
                flag = 1;
                cout << -1 << endl;
                break;
            }
        }
        if (!flag)
            cout << ans << endl;
    }
    return 0;
}

F - Fence Repair

题意：

有一个农夫要把一个木板钜成几块给定长度的小木板，每次锯都要收取一定费用，这个费用就是当前锯的这个木版的长度

给定各个要求的小木板的长度，及小木板的个数n，求最小费用

思路：

学了数据结构的同学应该不陌生，这是不是很像哈夫曼思想。
那么怎么来思考这道题目呢，我们首先能从题目得出锯一次木板所需的费用是锯断后的两块木板的长度之和。因此我们是不是可以得出这样一个结论就是先锯得所需的木板中较长的木板，因为如果还未锯取这块木板时，我们每次锯木块收取费用时都把他的价值加上了。

那怎么实现呢，我们可以逆向推到，用优先队列来存最后的木板即所需的木板，每次都把最小的两个相加再入队，最后队列里只剩一个数时即最原始的木板，这时所得的费用就是所要求的答案。

代码：

#include<iostream>
#include<queue>
#define ll long long
using namespace std;

int main() {
    
    int n, t;
    cin >> n;
    priority_queue <int ,vector<int>,greater<int> > que;//设一个小顶堆
    for(int i = 0; i < n; i++) {
        cin >> t;
        que.push(t);//入队
    }
    if(que.size() == 1) {//n为1时直接输出
        cout << que.top();
        return 0;
    }
    ll sum = 0;
    while(que.size() > 1) {
        ll minf, mins;//第一小和第二小
        minf = que.top();
        que.pop();
        mins = que.top();
        que.pop();
        sum += minf + mins;//记录花费
        que.push(minf + mins);//再入队
    }
    cout << sum << endl;
    return 0;
}

G - Throwing cards away I

题意：

给n张牌放成一叠，从上到下编号从1~n，当至少还有两张牌时，丢弃最上面的牌，第二张牌放到最下面，重复这个过程直到只剩下一张牌，输出丢弃牌的序列以及最后剩下的牌。

思路：

水题啦，直接模拟队列就能解决了

代码：

懒得写了，直接copy一份通过的代码过来了 

#include <algorithm>
#include <iostream>
#include <cstdlib>
#include <cstring>
#include <cstdio>
#include <cmath>
 
using namespace std;
 
int Q[10000];
 
int main()
{
    int n;
    while (cin >> n && n) {
        for (int i = 0; i <= n; ++ i)
            Q[i] = i;
        int head = 1,tail = n,first = 1;
        printf("Discarded cards:");
        while (head < tail) {
            if (!first) printf(",");
            printf(" %d",Q[head ++]);
            Q[++ tail] = Q[head ++];
            first = 0;
        }
        printf("\nRemaining card: %d\n",Q[head]);
    }
    return 0;
}

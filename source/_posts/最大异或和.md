# 异或的性质

<!-- more -->

1. a^a=0
2. a^b=b^a
3. a^b^a=b
4. a^b=!a^!b
5. !a^b=a^!b
6. 若s=a^b ,有a=s^b

令
$$
S_i=a_1⊕a_2⊕…⊕a_i
$$
则,由性质1
$$
a_l⊕a_{l+1}⊕…⊕a_r=S_r⊕S_{l-1}
$$

# Acwing 3485最大异或和

```cpp
#include<bits/stdc++.h>

using namespace std;

const int N=3100010;

int tr[N][2],s[N],t[N];
int n,m,ans=0,cnt=1;

void insert(int x,int v)
{
    int p=0;
    for(int i=30;i>=0;i--)
    {
        int u=(x>>i)&1;
        if(!tr[p][u])
            tr[p][u]=cnt++;
        p=tr[p][u];
        t[p]+=v;
    }
}

int query(int x)
{
    int p=0,res=0;
    for(int i=30;i>=0;i--)
    {
        int u=(x>>i) & 1;
        if(t[tr[p][!u]])
        {
            p = tr[p][!u];
            res=res*2+1;
        }
        else
            res=res*2,p=tr[p][u];
            
    }
    return res;
}

signed main()
{
    cin>>n>>m;
    for(int i=1;i<=n;i++)
    {
        int x;
        cin>>x;
        s[i]=s[i-1]^x;
    }
    insert(s[0],1);
    for(int i=1;i<=n;i++)
    {
        if(i-m-1>=0)
            insert(s[i-m-1],-1);
        ans=max(ans,query(s[i]));
        insert(s[i],1);
    }
    cout<<ans;
}
```

# [Codeforces Round 882 (Div. 2)](https://codeforces.com/contest/1847) C

```cpp
#include<bits/stdc++.h>

using namespace std;

const int N = 1e5+10;

int tr[N][2], s[N], num[N];//s是前缀和，num是这个前缀出现的次数

int t, n;
int cnt = 1;//把cnt=1留给根节点
int x;

void insert(int x)
{
	int p = 0;
	for (int i = 10; i >= 0; i--)
	{
		int u = (x >> i) & 1;
		if (!tr[p][u])
			tr[p][u] = cnt++;
		p = tr[p][u];
		num[p]++;
	}
}

int query(int x)
{
	int p = 0, res = 0;
	for (int i = 10; i >= 0; i--)
	{
		int u = (x >> i) & 1;
		if (num[tr[p][!u]])
		{
			p = tr[p][!u];
			res = res * 2 + 1;

		}
		else
			res = res * 2, p = tr[p][u];
	}
	return res;
}

signed main()
{
	cin >> t;
	while (t--)
	{
		s[0] = 0;
		memset(tr, 0, sizeof(tr));
		cnt = 1;
		int ans = 0;
		scanf_s("%d", &n);
		for (int i = 1; i <= n; i++)
		{
			cin >> x;
			s[i] = s[i - 1] ^ x;
		}
		insert(s[0]);
		for (int i = 1; i <= n; i++)
		{
			ans = max(ans, query(s[i]));
			insert(s[i]);
		}
		printf("%d\n", ans);
	}
}
```

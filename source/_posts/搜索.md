# Flood Fill

<!-- more -->

可以在线性时间复杂度内，找到某个点所在的连通块

Acwing 1097

```cpp
#include<bits/stdc++.h>

using namespace std;

const int N = 1e3 + 10;

typedef pair<int, int> PII;

char s[N][N];
bool st[N][N];
int n, m;
int cnt = 0;

void bfs(int x, int y)
{
    queue<PII> q;
    q.push({ x,y });
    while (!q.empty())
    {
        PII p = q.front();
        q.pop();
        for (int i = p.first - 1; i <= p.first + 1; i++)
        {
            for (int j = p.second - 1; j <= p.second + 1; j++)
            {
                if (i == p.first && j == p.second)
                    continue;
                if (i < 0 || j < 0 || i >= n || j >= m)
                    continue;
                if (s[i][j] == '.' || st[i][j] == 1)
                    continue;
                q.push({ i,j });
                st[i][j] = true;
            }
        }

    }
}

signed main()
{
    cin >> n >> m;
    for (int i = 0; i < n; i++)
    {
        for (int j = 0; j < m; j++)
            cin >> s[i][j];
    }

    for (int i = 0; i < n; i++)
    {
        for (int j = 0; j < m; j++)
        {
            if (s[i][j] == 'W' && st[i][j] == 0)
            {
                bfs(i, j);
                cnt++;
            }
        }
    }
    cout << cnt;
}
```

```cpp
#include<bits/stdc++.h>

using namespace std;

typedef pair<int, int> PII;

const int N = 1e3 + 10;

int n, m;
int dist[N][N];
int dir[4][2] = { 0,1,0,-1,1,0,-1,0 };
char a[N][N];

void bfs()
{
	queue<PII> q;
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < m; j++)
		{
			dist[i][j] = -1;
			if (a[i][j] == '1')
				q.push({ i,j }),dist[i][j]=0;
		}
	}

	while (!q.empty())
	{
		PII p = q.front();
		q.pop();
		for (int i = 0; i < 4; i++)
		{
			int tx = p.first + dir[i][0], ty = p.second + dir[i][1];
			if (tx < 0 || tx >= n || ty < 0 || ty >= m)
				continue;
			if (dist[tx][ty] != -1)
				continue;
			q.push({ tx,ty });
			dist[tx][ty] = dist[p.first][p.second] + 1;
		}
	}
}

signed main()
{
	cin >> n >> m;
	for (int i = 0; i < n; i++)
		cin >> a[i];
	bfs();
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < m; j++)
			cout << dist[i][j] << ' ';
		cout << '\n';
	}
}
```

```cpp
#include <cstring>
#include <algorithm>
#include <unordered_map>

using namespace std;

typedef pair<int, int> PII;
const int N = 1e6;

int g[2][4];

string start,tar;//@不要用end命名

unordered_map<string,int>dist;//到string状态用的步数
unordered_map<string,pair<char,string>>pre;// 变化到键的状态的操作是char存储，键状态的前一个状态是值里的string存贮

string q[N];//队列

string move0(string st){
    string res = st;
    reverse(res.begin(),res.end());
    return res;
}

string move1(string t)
{
    for(int i=0;i<3;i++) swap(t[3],t[i]);
    for(int i=4;i<7;i++) swap(t[i],t[i+1]);
    return t;
}

string move2(string st){
    string res = st;
    swap(st[1],st[6]);
    swap(st[5],st[6]);
    swap(st[2],st[5]);
    return st;
}

int bfs(){
    if(start == tar)return 0;
    q[0] = start;
    int hh = 0,tt = 0;
    dist[start] = 0;
    while(hh <= tt){
        string t = q[hh ++];
        string m[3];
        m[0] = move0(t);
        m[1] = move1(t);
        m[2] = move2(t);
        //按照ABC三种操作的顺序BFS正好能找到最小字典序
        //因为bfs的过程就保证了是按最小字典序找的，A操作的后续必然先扩展，先加入队列，之后是B,C

        //下面看的都是将要转移的状态
        for(int i = 0; i < 3; ++ i){
            if(!dist.count(m[i])){//状态没有被访问过

                dist[m[i]] = dist[t] + 1;
                pre[m[i]] = {'A' + i,t};//m[i]由A + i操作转移过来
                q[++ tt] = m[i];
                if(m[i] == tar) return dist[tar];
            }
        }
    }
    return -1;
}

    for(int i = 0; i < 8; ++ i){

    int step = bfs();

    while(tar != start){
        res += pre[tar].first;//取操作（逆序）
        tar = pre[tar].second;//变上一个状态
    }
    reverse(res.begin(),res.end());
    if(step > 0)cout << res << endl;

    return 0;
}
```

```cpp
#include<bits/stdc++.h>
#include<unordered_map>

using namespace std;

#define x first
#define y second

typedef pair<int, string> PIS;

int dir[4][2] = { 0,1,0,-1,1,0,-1,0 };

//估价函数：因为每次移动最多向目标位置移动一格，
//所以估价函数等于真实位置与目标位置的曼哈顿距离之和
int f(string state)//预估代价
{
	int res = 0;
	for (int i = 0; i < state.size(); i++)
	{
		if (state[i] != 'x')
		{
			int t = state[i] - '1';
			res += abs(i / 3 - t / 3) + abs(i % 3 - t % 3);
		}
	}
	return res;
}

string bfs(string start)
{
	string endd = "12345678x";
	char op[] = "rldu";

	unordered_map<string, int>dist;
	//上一步操作
	unordered_map<string, pair<char, string>> prev;
	priority_queue<PIS, vector<PIS>, greater<PIS>> heap;

	dist[start] = 0;
	heap.push({ f(start),start });

	while (heap.size())
	{
		auto t = heap.top();
		heap.pop();

		string state = t.y;
		if (state == endd)
			break;

		int x, y;
		for (int i = 0; i < 9; i++)
		{
			if (state[i] == 'x')
			{
				x = i / 3, y = i % 3;
				break;
			}
		}

		string source = state;
		for (int i = 0; i < 4; i++)
		{
			int a = x + dir[i][0], b = y + dir[i][1];
			if (a < 0 || a>2 || b < 0 || b>2)
				continue;
			state = source;
			swap(state[x * 3 + y], state[a * 3 + b]);
			if (dist.count(state) == 0 || dist[state] > dist[source]+1)
			{
				dist[state] = dist[source] + 1;
				prev[state] = { op[i],source };
				heap.push({ dist[state] + f(state),state });
			}
		}
	}

	string res;
	while (endd != start)
	{
		res += prev[endd].x;
		endd = prev[endd].y;
	}
	reverse(res.begin(), res.end());
	return res;
}

signed main()
{
	string seq, start;
	char c;
	while (cin >> c)
	{
		start += c;
		if (c != 'x')
			seq += c;
	}

	int cnt = 0;//逆序对数量
	//求逆序对数量
	for (int i = 0; i < 8; i++)
	{
		for (int j = i; j < 8; j++)
		{
			if (seq[i] > seq[j])
				cnt++;
		}
	}
	//因为A*算法实现的条件之一是必须能够实现，而只有逆序对数量是偶数才能实现
	if (cnt & 1)
		puts("unsolvable");
	else
		cout << bfs(start) << endl;
}
```

# DFS之迭代加深

[170. 加成序列 - AcWing题库](https://www.acwing.com/problem/content/description/172/)

```cpp
#include<bits/stdc++.h>

using namespace std;

const int N = 1e4 + 10;

int n;
int dep[N];

bool dfs(int u, int depth)
{
	if (u > depth)
		return false;
	if (dep[u - 1] == n)
		return true;
	bool st[N] = { 0 };
	for (int i = u - 1; i >= 0; i--)
	{
		for (int j = i; j >= 0; j--)
		{
			int s = dep[i] + dep[j];
			if (s > n || s <= dep[u - 1] || st[s])
				continue;
			st[s] = true;
			dep[u] = s;
			if (dfs(u + 1, depth))
				return true;
		}
	}
	return false;
}

signed main()
{
	dep[0] = 1;
	while (cin >> n , n)
	{
		int depth = 1;
		while (!dfs(1, depth))
			depth++;
		for (int i = 0; i < depth; i++)
			cout << dep[i] << ' ';
		cout << '\n';
	}
}
```

# 双向DFS

[171. 送礼物 - AcWing题库](https://www.acwing.com/problem/content/description/173/)

```cpp
#include<bits/stdc++.h>
#define int long long

using namespace std;

const int N = 50;

int a[N];
int weight[1 << 25];//k最大是25， 因此最多可能有2^25种方案
int n, m,k,cnt;
int ans;

void dfs1(int u, int sum)
{
	if (u > k)
	{
		// 如果我们当前已经枚举完第k个数了， 
		//就把当前的sum， 加到weights中去
		weight[++cnt] = sum;
		return;
	}
	// 枚举当前不选这个物品
	dfs1(u + 1, sum);
	// 选这个物品, 做一个可行性剪枝
	if (sum + a[u] <= m)
		dfs1(u + 1, sum + a[u]);
}

void dfs2(int u, int sum)
{
	if (u > n)
	{
		// 如果已经找完了n个节点， 那么需要二分一下
		int l = 1, r = cnt;
		while (l < r)
		{
			int mid = l + r + 1 >> 1;
			if (weight[mid] + sum <= m)
				l = mid;
			else
				r = mid - 1;
		}
		ans = max(ans, weight[l] + sum);
		return;
	}

	dfs2(u + 1, sum);
	if (sum + a[u] <= m)
		dfs2(u + 1, sum + a[u]);
}

signed main()
{
	cin >> m >> n;
	for (int i = 1; i <= n; i++)
		cin >> a[i];
	sort(a+1, a + n+1, greater<int>());

	k = n / 2;
	dfs1(1, 0);
	sort(weight + 1, weight + cnt + 1);
	cnt = unique(weight + 1, weight + 1 + cnt) - weight - 1;
	dfs2(k + 1, 0);
	cout << ans << '\n';

}
```

# IDA*

[180. 排书 - AcWing题库](https://www.acwing.com/problem/content/description/182/)

```cpp
#include <iostream>
#include <cstring>
using namespace std;
const int N = 20;
int n;
int a[N];
int w[5][N];
int f () {
    int ans = 0;
    for (int i = 2;i <= n;i++) ans += a[i - 1] + 1 != a[i];
    return (ans + 2) / 3;
}
bool dfs (int depth,int max_depth) {
    if (depth + f () > max_depth) return false;
    if (f () == 0) return true;
    for (int len = 1;len <= n;len++) {
        for (int l = 1;l + len - 1 <= n;l++) {
            int r = l + len - 1;
            for (int k = r + 1;k <= n;k++) {
                memcpy (w[depth],a,sizeof (a));
                int x,y;
                for (x = r + 1,y = l;x <= k;x++,y++) a[y] = w[depth][x];
                for (x = l;x <= r;x++,y++) a[y] = w[depth][x];
                if (dfs (depth + 1,max_depth)) return true;
                memcpy (a,w[depth],sizeof (w[depth]));
            }
        }
    }
    return false;
}
int main () {
    int T;
    cin >> T;
    while (T--) {
        cin >> n;
        for (int i = 1;i <= n;i++) cin >> a[i];
        int max_depth = 0;
        while (max_depth < 5 && !dfs (0,max_depth)) max_depth++;
        if (max_depth == 5) puts ("5 or more");
        else cout << max_depth << endl;
    }
    return 0;
}
```

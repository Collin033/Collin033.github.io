## AcWing 1027. 方格取数

**历史图片缺失：image-20230519235343338（原图片未进入 Git 仓库，现已移除失效链接）**

```cpp
#include<bits/stdc++.h>

using namespace std;

const int N = 30;

int w[N][N];
int f[N][N][N];

int n;

signed main()
{
	cin >> n;
	while (1)
	{
		int a, b, c;
		cin >> a >> b >> c;
		if (a == 0 && b == 0 && c == 0)
			break;
		w[a][b] = c;
	}
	for (int k = 2; k <= 2 * n; k++)
	{
		for (int i1 = 1; i1 <= n; i1++)
		{
			for (int i2 = 1; i2 <= n; i2++)
			{
				int j1 = k - i1, j2 = k - i2;
				if (j1 >= 1 && j1 <= n && j2 >= 1 && j2 <= n)
				{
					int t = w[i1][j1];
					if (i1 != i2)
						t += w[i2][j2];
					int& x = f[k][i1][i2];
					x = max(x, f[k - 1][i1 - 1][i2 - 1]+t);
					x = max(x, f[k - 1][i1 ][i2 - 1]+t);
					x = max(x, f[k - 1][i1 - 1][i2]+t);
					x = max(x, f[k - 1][i1][i2 ]+t);
				}
			}
		}
	}
	cout << f[n+n][n][n];
}
```

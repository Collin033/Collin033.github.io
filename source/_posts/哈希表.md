## 一般哈希 —— 模板题 AcWing 840. 模拟散列表

### 拉链法

```cpp
#include<bits/stdc++.h>
#define int long long
using namespace std;

const int N = 100003;

char s[2];
int n,x;
int ne[N], h[N], e[N], idx;

void ins(int x)
{
	int k = (x % N + N) % N;
	e[idx] = x;
	ne[idx] = h[k];
	h[k] = idx++;
}

bool findd(int x)
{
	int k = (x % N + N) % N;
	for (int i = h[k]; i != -1; i = ne[i])
	{
		int j = e[i];
		if (j == x)
			return true;
	}
	return false;
}

signed main()
{
	memset(h, -1, sizeof(h));
	cin >> n;
	while (n--)
	{
		cin >> s >> x;
		if (s[0] == 'I')
			ins(x);
		else
		{
			if (findd(x))
				puts("Yes");
			else
				puts("No");
		}
	}
}
```

### 开放寻址法

```cpp
#include<bits/stdc++.h>
#define int long long
using namespace std;

const int N = 100003;

char s[2];
int n,x;
int h[N];


int findd(int x)
{
	int k = (x % N + N) % N;
	while (h[k] != -1 && h[k] != x)
	{
		k++;
		if (k == N)
			return 0;
	}
	return k;
}

signed main()
{
	memset(h, -1, sizeof(h));
	cin >> n;
	while (n--)
	{
		cin >> s >> x;
		if (s[0] == 'I')
			h[findd(x)] = 1;
		else
		{
			if (h[findd(x)]==1)
				puts("Yes");
			else
				puts("No");
		}
	}
}
```

## 字符串哈希 —— 模板题 AcWing 841. 字符串哈希

**核心思想**：将字符串看成P进制数，P的经验值是131或13331，取这两个值的冲突概率低
**小技巧：**取模的数用2^64，这样直接用unsigned long long存储，溢出的结果就是取模的结果

```text
p数组用来存储h要乘的p的多少次方

h数组用来存储每个前缀字符串的哈希值
    如str = "ABCD"
    h[0] = 0;
    h[1] = "A"的哈希值（将字符转换成p进制的一个数字）
    h[2] = "AB"的哈希值（将字符转换成p进制的一个数字）
    h[3] = "ABC"的哈希值（将字符转换成p进制的一个数字）
    h[4] = "ABCD"的哈希值（将字符转换成p进制的一个数字）
```

```text
把前l-1位乘上p的r-l+1次方（相当于对齐位数并补0），再用h[r]减去这个数即可

      举个例子：

有一个字符串ABCDE ，下标从0开始
      
      输入l = 1, r = 3
      
      假设哈希值计算出来如下：
      
      h[3] = 1234
      h[1] = 12
      h[0] = 1
      
      则根据公式：h[r] - h[l - 1] * p[r - l + 1]
      
      得到：
      h[3] - h[1 - 1] * p[3 - 1 + 1] = 1234 - 1*1000 = 234
      
      即这个234就表示了[l,r]的哈希值（注意是左闭右闭）
      
      p[i]数组表示p的i次方的值是多少
```

```cpp
#include<bits/stdc++.h>
#define int long long
typedef unsigned long long ULL;
using namespace std;

const int N = 1e5 + 10;
const int P = 131;

ULL h[N], p[N];

int n, m;
char str[N];

ULL gett(int l, int r)
{
	return h[r] - h[l - 1] * p[r - l + 1];
}

signed main()
{
	cin >> n >> m>>str+1;
	p[0] = 1;

	for (int i = 1; i <= n; i++)
	{
		p[i] = p[i - 1] * P;
		h[i] = h[i - 1] * P + str[i];
	}

	while (m--)
	{
		int l1, r1, l2, r2;
		cin >> l1 >> r1 >> l2 >> r2;
		if (gett(l1, r1) == gett(l2, r2))
			puts("Yes");
		else
			puts("No");
		
	}
	return 0;

}
```

## 求逆序对数量AcWing 788. 逆序对的数量

```cpp
#include<bits/stdc++.h>
#define int long long

using namespace std;

const int N=1e5+10;

int a[N],b[N];
int n,cnt=0;

void merge(int l,int mid,int r)
{
    int i=l,t=0,j=mid+1;
    while(i<=mid && j<=r)
    {
        if(a[i]>a[j])
        {
            b[++t]=a[j++];
            cnt+=mid-i+1;
        }
        else
            b[++t]=a[i++];
    }
    while(i<=mid)
        b[++t]=a[i++];
    while(j<=r)
        b[++t]=a[j++];
    for(int i=1;i<=t;i++)
        a[l+i-1]=b[i];
}

void mergesort(int l,int r)
{
    if(l<r)
    {
        int mid=(l+r)>>1;
        mergesort(l,mid);
        mergesort(mid+1,r);
        merge(l,mid,r);
    }
}

signed main()
{
    cin>>n;
    for(int i=1;i<=n;i++)
        cin>>a[i];
    mergesort(1,n);
    cout<<cnt;
}
```

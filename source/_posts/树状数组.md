# [241. 楼兰图腾 - AcWing题库](https://www.acwing.com/problem/content/243/) (单点插入，区间查询)

<!-- more -->

```cpp
#include<bits/stdc++.h>
#define int long long

using namespace std;

const int N = 2e5 + 10;

int a[N];
int Greater[N], Lower[N];
int tr[N];
int n;

int lowbit(int x)
{
	return x & -x;
}

void add(int x, int c)
{
	for (int i = x; i <= n; i += lowbit(i))
		tr[i] += c;
}

int summ(int x)
{
	int res = 0;
	for (int i = x; i; i -= lowbit(i))
		res += tr[i];
	return res;
}

signed main()
{
	cin >> n;
	for (int i = 1; i <= n; i++)
		cin >> a[i];
	for (int i = 1; i <= n; i++)
	{
		int x = a[i];
		Greater[i] = summ(n) - summ(x);//x左边比x大的数
		Lower[i] = summ(x - 1);//x左边比x小的数
		add(x, 1);//在当前位置插入1
	}
	memset(tr, 0, sizeof(tr));
	int res1 = 0, res2 = 0;
	for (int i = n; i; i--)
	{
		int x = a[i];
		res1 += Greater[i] * (summ(n) - summ(x));//x右边比x大的数
		res2 += Lower[i] * (summ(x - 1));//x右边比x小的数
		add(x, 1);
	}
	cout << res1 << ' ' << res2;
}
```

# [242. 一个简单的整数问题 - AcWing题库](https://www.acwing.com/problem/content/description/248/) (区间插入，单点查询)

```cpp
#include<bits/stdc++.h>
#define int long long

using namespace std;

const int N = 1e5 + 10;

int a[N];
int tr[N];
int n,m;

int lowbit(int x)
{
	return x & -x;
}

void add(int x, int c)
{
	for (int i = x; i <= n; i += lowbit(i))
		tr[i] += c;
}

int summ(int x)
{
	int res = 0;
	for (int i = x; i; i -= lowbit(i))
		res += tr[i];
	return res;
}

signed main()
{
	cin >> n >> m;
	for (int i = 1; i <= n; i++)
		cin >> a[i];
	for (int i = 1; i <= n; i++)
		add(i, a[i] - a[i - 1]);

	while (m--)
	{
		char op[2];
		int l, r, d;
		cin >> op >> l;
		if (*op == 'C')
		{
			cin >> r >> d;
			add(l, d);
			add(r + 1, -d);
		}
		else
		{
			cout << summ(l) << '\n';
		}
	}
}
```

# [243. 一个简单的整数问题2 - AcWing题库](https://www.acwing.com/problem/content/description/244/)(区间修改，区间查询)

```cpp
#include<bits/stdc++.h>
#define int long long

using namespace std;

const int N = 1e5 + 10;

int a[N];
int n, m;
int tr1[N];
int tr2[N];

int lowbit(int x)
{
	return x & -x;
}

void add(int tr[], int x, int c)
{
	for (int i = x; i <= n; i += lowbit(i))
		tr[i] += c;
}

int summ(int tr[], int x)
{
	int res = 0;
	for (int i = x; i; i -= lowbit(i))
		res += tr[i];
	return res;
}

int presum(int x)
{
	return summ(tr1, x) * (x + 1) - summ(tr2, x);
}


signed main()
{
	cin >> n >> m;
	for (int i = 1; i <= n; i++)
		cin >> a[i];
	for (int i = 1; i <= n; i++)
	{
		int x = a[i] - a[i - 1];
		add(tr1, i, x);
		add(tr2, i, x * i);
	}

	while (m--)
	{
		char op[2];
		int l, r, d;
		cin >> op >> l >> r;
		if (*op == 'C')
		{
			cin >> d;
			add(tr1, l, d);
			add(tr1, r + 1, -d);
			add(tr2, l, l * d);
			add(tr2, r + 1, (r + 1) * (-d));
		}
		else
			cout << presum(r) - presum(l - 1) << '\n';
	}
}
```

# [244. 谜一样的牛 - AcWing题库](https://www.acwing.com/problem/content/description/245/)

```cpp
#include<bits/stdc++.h>
#define int long long

using namespace std;

const int N = 1e5 + 10;

int a[N],ans[N];
int n;
int tr[N];

int lowbit(int x)
{
	return x & -x;
}

void add(int x, int c)
{
	for (int i = x; i <= n; i += lowbit(i))
		tr[i] += c;
}

int summ(int x)
{
	int res = 0;
	for (int i = x; i; i -= lowbit(i))
		res += tr[i];
	return res;
}

signed main()
{
	cin >> n;
	for (int i = 2; i <= n; i++)
		cin >> a[i];

	for (int i = 1; i <= n; i++)
		tr[i] = lowbit(i);

	for (int i = n; i; i--)
	{
		int k = a[i] + 1;
		int l = 1, r = n;
		while (l < r)
		{
			int mid = l + r >> 1;
			if (summ(mid) >= k)
				r = mid;
			else
				l = mid + 1;
		}
		ans[i] = l;
		add(l, -1);
	}
	for (int i = 1; i <= n; i++)
		cout << ans[i] << '\n';
}
```

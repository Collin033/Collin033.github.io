```cpp
using namespace std;

	cin >> n >> m;

	for (int i = 1; i <= n; i++)

	{
		for (int j = 0; j <= m; j++)
		{
			f[i][j] = f[i - 1][j];
			if (j >= v[i])
				f[i][j] = max(f[i - 1][j], f[i - 1][j - v[i]] + w[i]);
		}
	}

	cout << f[n][m];
}
```

### 优化版 Acwing 5（二进制优化）

```cpp
#include<bits/stdc++.h>

using namespace std;

const int N = 3e4+5;

int n, m;
int v[N], w[N],s[N];
int f[N];

signed main()
{
	cin >> n >> m;

	int cnt = 0;

	for (int i = 1; i <= n; i++)
	{
		int a, b, s;
		cin >> a >> b >> s;
		int k = 1;
		while (k <= s)
		{
			cnt++;
			v[cnt] = a * k;
			w[cnt] = b * k;
			s -= k;
			k *= 2;
		}
		if (s > 0)
		{
			cnt++;
			v[cnt] = a * s;
			w[cnt] = b * s;
		}
	}

	n = cnt;

	for (int i = 1; i <= n; i++)
		for (int j = m; j >= v[i]; j--)
			f[j] = max(f[j], f[j - v[i]] + w[i]);

	cout << f[m];
}
```

#### 二维

```cpp
#include<bits/stdc++.h>

using namespace std;

const int N = 1e3 + 10,M = 20010;

int n, m;
int v[N], w[N], s[N];
int f[N][M];
int q[M];


signed main()
{
	cin >> n >> m;
	for (int i = 1; i <= n; i++)
		cin >> v[i] >> w[i] >> s[i];

	for (int i = 1; i <= n; i++)
	{
		for (int r = 0; r < v[i]; r++)
		{
			int hh = 0, tt = -1;
			for (int j = r; j <= m; j += v[i])
			{
				while (hh <= tt && j - q[hh] > s[i] * v[i])
					hh++;
				while (hh <= tt && f[i - 1][q[tt]] + (j - q[tt]) / v[i] * w[i] <= f[i - 1][j])
					tt--;
				q[++tt] = j;
				f[i][j] = f[i - 1][q[hh]] + (j - q[hh]) / v[i] * w[i];
			}
		}
	}
	cout << f[n][m] << '\n';
}
```

### [石子合并](https://www.luogu.com.cn/problem/P1880)

```cpp
#include <bits/stdc++.h>
#define int long long

using namespace std;

const int N = 410,INF=0x3f3f3f3f;

int n;
int a[N],s[N];
int f[N][N], g[N][N];

signed main() 
{
    cin >> n;
    for (int i = 1; i <= n; i++)
    {
        cin >> a[i];
        a[i + n] = a[i];
    }

    for (int i = 1; i <= 2*n; i++)
        s[i] = s[i - 1] + a[i];

    memset(f, -0x3f, sizeof(f));
    memset(g, 0x3f, sizeof(g));

    for (int len = 1; len <= n; len++)
    {
        for (int l = 1; l + len-1 <= 2*n; l++)
        {
            int r = l + len-1;
            if (len == 1)
                f[l][r] = g[l][r] = 0;
            else
            {
                for (int k = l; k < r; k++)
                {
                    f[l][r] = max(f[l][r], f[l][k] + f[k + 1][r] + s[r] - s[l - 1]);
                    g[l][r] = min(g[l][r], g[l][k] + g[k + 1][r] + s[r] - s[l - 1]);
                }
            }

        }
    }

    int minn = INF, maxn = -INF;
    for (int i = 1; i <= n; i++)
    {
        minn = min(minn, g[i][i + n - 1]);
        maxn = max(maxn, f[i][i + n - 1]);
    }
    cout << minn << '\n' << maxn;

    return 0;
}
```

### [加分二叉树](https://www.luogu.com.cn/problem/P1040)

```cpp
#include <bits/stdc++.h>
#define int long long

using namespace std;

const int N = 31;

int n;
int w[N];
int f[N][N];
int g[N][N];

void dfs(int l, int r)
{
    if (l > r)
        return;
    int root = g[l][r];
    cout << root << ' ';
    dfs(l, root - 1);
    dfs(root + 1, r);
}

signed main() 
{
    cin >> n;
    for (int i = 1; i <= n; i++)
        cin >> w[i];

    for (int len = 1; len <= n; len++)
    {
        for (int l = 1; l + len - 1 <= n; l++)
        {
            int r = l + len - 1;
            if (len == 1)
            {
                f[l][r] = w[l];
                g[l][r] = l;
            }
            else
            {
                for (int k = l; k <= r; k++)
                {
                    int left = k == l ? 1 : f[l][k - 1];
                    int right = k == r ? 1 : f[k + 1][r];
                    int score = left * right + w[k];
                    if (f[l][r] < score)
                    {
                        f[l][r] = score;
                        g[l][r] = k;
                    }
                }
            }
        }
    }

    cout << f[1][n]<<'\n';
    dfs(1, n);
    return 0;
}
```

```cpp
#include<bits/stdc++.h>
#define int long long
using namespace std;


const int N = 15;
int ten[N], dp[N];
int cnta[N], cntb[N];
int num[N];

void init()
{
	ten[0] = 1;
	for (int i = 1; i <= N; i++)
	{
		dp[i] = i * ten[i - 1];
		ten[i] = 10 * ten[i - 1];
	}
}

void solve(int x, int* cnt)
{
	int len = 0;
	while (x)
	{
		num[++len] = x % 10;
		x /= 10;
	}
	for (int i = len; i >= 1; i--)
	{
		for (int j = 0; j <= 9; j++)
			cnt[j] += dp[i - 1] * num[i];
		for (int j = 0; j < num[i]; j++)
			cnt[j] += ten[i - 1];
		int num2 = 0;
		for (int j = i - 1; j >= 1; j--)
			num2 = num2 * 10 + num[j];
		cnt[num[i]] += num2 + 1;
		cnt[0] -= ten[i - 1];
	}
}

signed main()
{
	init();
	int a, b;
	while(cin>>a>>b && (a && b))
	{
	    memset(cnta,0,sizeof(cnta));
	    memset(cntb,0,sizeof(cntb));
	    if(a>b) swap(a,b);
    	solve(a-1, cnta); solve(b, cntb);
    	for (int i = 0; i <= 9; i++)
    	{
    		cout << cntb[i] - cnta[i] << ' ';
    	}
    	cout<<'\n';
	}
	
}
```

### AcWing 1081. 度的数量

```cpp
#include <bits/stdc++.h>
#define int long long

using namespace std;

const int N = 35;

int c[N][N];//存储组合数
int K, B;

//预处理组合数c[a][b]表示Cab
void init()
{
    for (int i = 0; i < N; i++)
    {
        for (int j = 0; j <= i; j++)
        {
            if (!j)
                c[i][j] = 1;
            else
                c[i][j] = c[i - 1][j] + c[i - 1][j - 1];
        }
    }
}

int dp(int n)
{
    if (!n)
        return 0;
    vector<int> num;
    while (n)
        num.push_back(n % B), n /= B;

    int ans = 0, one = 0;//one表示已经填了多少个1

    for (int i = num.size() - 1; i >= 0; i--)
    {
        int t = num[i];
        //若t=0,那t只能等于0,到下一次继续进行划分
        if (t)//对第i位进行讨论
        {
            ans += c[i][K - one];//第i位填0，剩下i-1位填K-one个1
            //第i位填1
            if (t > 1)
            {
                //第i位填1,剩下的i-1位填剩余个1
                if (K - one - 1 >= 0)
                    ans += c[i][K - one - 1];
                break;
            }
            else
            {
                one++;//t等于1,剩下的位次不能乱填，所以再进行讨论划分
                if (one > K)
                    break;
            }
            //到底也算一种方案
            if (!i && one == K)
                ans++;
        }
    }
    return ans;
}

signed main() 
{
    int x, y;
    cin >> x >> y;
    init();

    cout<<dp(y) - dp(x - 1);
    return 0;
}
```

### AcWing 1082. 数字游戏

```cpp
#include <bits/stdc++.h>
#define int long long

using namespace std;

const int N = 15;

int f[N][N];//f[i][j]表示一共有i位，且最高位填j

void init()
{
    for (int i = 0; i <= 9; i++)
        f[1][i] = 1;

    for (int i = 2; i < N; i++)
    {
        for (int j = 0; j <= 9; j++)
        {
            for (int k = j; k <= 9; k++)
                f[i][j] += f[i - 1][k];
        }
    }
}

int dp(int n)
{
    if (!n)
        return 1;

    vector<int> num;
    while (n)
        num.push_back(n % 10), n /= 10;

    int res = 0, last = 0;

    for (int i = num.size() - 1; i >= 0; i--)
    {
        int x = num[i];
        //第i位填last~x-1
        for (int j = last; j < x; j++)
            res += f[i + 1][j];

        if (x < last)
            break;
        //第i位填x，继续循环，树的右子树
        last = x;

        //取到最后一位,按左子树选择
        if (!i)
            res++;
    }
}

signed main() 
{
    init();

    int l, r;
    while (cin >> l >> r)
        cout << dp(r) - dp(l - 1) << endl;
    return 0;
}
```

#### 朴素版

```cpp
#include<bits/stdc++.h>
#define int long long

using namespace std;

const int N = 12, M = 1 << N;

bool st[M];
int f[N][M];

int n, m;

signed main()
{
	while (cin >> n >> m && (n || m))
	{
		for (int i = 0; i < (1 << n); i++)
		{
			st[i] = 1;
			int cnt = 0;
			for (int j = 0; j < n; j++)
			{
				if (i >> j & 1)
				{
					if (cnt & 1)
					{
						st[i] = false;
						break;
					}
				}
				else
					cnt++;
			}
			if (cnt & 1)
				st[i] = false;
		}

		memset(f, 0, sizeof(f));
		f[0][0] = 1;

		for (int i = 1; i <= m; i++)
		{
			for (int j = 0; j < 1 << n; j++)
			{
				for (int k = 0; k < 1 << n; k++)
				{
					if (!(j & k) && (st[j | k]))
						f[i][j] += f[i - 1][k];
				}
			}
		}

		cout << f[m][0] << '\n';
	}
}
```

#### 优化版

```cpp
#include <iostream>
#include <vector>
#include <cstring>
using namespace std;

const int N = 12, M = 1 << N;

bool st[M];

long long f[N][M];

int n, m;

vector<int> vs[M]; // 对于状态j, vs[j] 存储的是一个数组, 数组中每个元素, 都是能够转移到 j 的有效状态

int main() {
	while(true) {
		scanf("%d%d", &n, &m);
		if(n == 0 && m == 0) break;

		// 预处理
		for(int i = 0; i < 1 << n; i++) {
			st[i] = true;
			int cnt = 0;
			for(int j = 0; j < n; j++) {
				if(i >> j & 1) {
					if(cnt & 1) {
						st[i] = false;
						break;
					}
				} else cnt++;
			}
			if(cnt & 1) st[i] = false;
		}

		for(int i = 0; i < 1 << n; i++) {
			vs[i].clear(); //清除, 因为要重复使用
			for(int k = 0; k < 1 << n; k++) {
				if((i & k) == 0 && st[i | k]) vs[i].push_back(k);
			}
		}

		memset(f, 0, sizeof f);
		f[0][0] = 1;
		for(int i = 1; i <= m; i++) {
			for(int j = 0; j < 1 << n; j++) {
				for(auto k : vs[j]) { // 直接遍历有效状态, 进行累加即可
					f[i][j] += f[i - 1][k];
				}
			}
		}

		printf("%lld\n", f[m][0]);
	}
	return 0;
}
```

### [最短Hamilton路径]([91. 最短Hamilton路径 - AcWing题库](https://www.acwing.com/problem/content/93/))

```cpp
#include<bits/stdc++.h>


using namespace std;

int n;

{
	memset(dp, 0x3f3f3f3f, sizeof(dp));
```

```cpp
    }
    
    m = state.size();

    for (auto i:state)
    {
        for (auto j:state)
        {
            if (!(i & j) && check(i | j))
```

### AcWing 1072. 树的最长路径【树形DP】

```cpp
#include <bits/stdc++.h>
#define int long long

using namespace std;

const int N = 1e4 + 10, M = N << 1;

int n;
int h[N], ne[N], w[N], e[N], idx;
int res;

void add(int a, int b, int c)
{
    e[idx] = b, w[idx] = c, ne[idx] = h[a], h[a] = idx++;
}

int dfs(int u, int father)
{
    int dist = 0;//从当前点往下走的最大长度
    int d1 = 0, d2 = 0;

    for (int i = h[u]; i != -1; i = ne[i])
    {
        int j = e[i];
        if (j == father)continue;
        int d = dfs(j, u) + w[i];
        dist = max(dist, d);

        if (d > d1)
            d2 = d1, d1 = d;
        else if (d > d2)
            d2 = d;
    }

    res = max(res, d1 + d2);
    return dist;

}

signed main() 
{
    memset(h, -1, sizeof h);
    cin >> n;
    for (int i = 1; i < n; i++)
    {
        int a, b, c;
        cin >> a >> b >> c;
        add(a, b, c);
        add(b, a, c);
    }

    dfs(1, -1);
    cout << res;
    return 0;
}
```

### [没有上司的舞会](https://www.luogu.com.cn/problem/P1352)

```cpp
#include <bits/stdc++.h>
#define int long long

using namespace std;

const int N = 6e3 + 10;

int n;
int ne[N], h[N], e[N], idx;
int happy[N];
bool has_father[N];
int dp[N][2];

void add(int a, int b)
{
    e[idx] = b, ne[idx] = h[a], h[a] = idx++;
}

void dfs(int u)
{
    dp[u][1] = happy[u];
    for (int i = h[u]; i != -1; i = ne[i])
    {
        int j = e[i];
        dfs(j);
        dp[u][1] += dp[j][0];
        dp[u][0] += max(dp[j][0], dp[j][1]);
    }
}

signed main() 
{
    memset(h, -1, sizeof h);
    cin >> n;
    for (int i = 1; i <= n; i++)
        cin >> happy[i];
    for (int i = 1; i < n; i++)
    {
        int a, b;
        cin >> a >> b;
        add(b,a);
        has_father[a] = true;
    }

    int root = 1;
    while (has_father[root])
        root++;
    dfs(root);
    cout << max(dp[root][0], dp[root][1]);
    return 0;
}
```
